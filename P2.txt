
#include <iostream>
#include <cstdlib>


using namespace std;

		struct nodo{
 int nro;
   			  struct nodo *izq, *der;
}

	typedef struct nodo *ABB;
// es un puntero de tipo nodo que hemos llamado ABB, que ulitizaremos
   //para mayor facilidad de creacion de variables 

ABB crearNodo(int x)
{
     ABB nuevoNodo = new(struct nodo);
     nuevoNodo->nro = x;


    		 nuevoNodo->izq = NULL;
      nuevoNodo->der = NULL;

     return nuevoNodo;
}
